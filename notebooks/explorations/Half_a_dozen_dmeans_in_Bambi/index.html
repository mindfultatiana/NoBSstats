import pymc as pm
import pytensor

def ShiftedExp(name, lam,  **kwargs):
    def shited_exp(lam, loc, size):
        return pm.Exponential.dist(lam=lam) + loc
    return pm.CustomDist(name, lam, 1, dist=shited_exp, **kwargs)
